function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Data Tes Drive") || ss.insertSheet("Data Tes Drive");
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(["Timestamp","Nama Driver","Jenis SIM","Nama Penguji",
      "NIK Penguji","Tanggal Tes","Aspek 1 - Pedal&Transmisi",
      "Aspek 2 - Kemudi&Kontrol","Aspek 3 - Pengetahuan&Safety",
      "Aspek 4 - Persiapan","Aspek 5 - Konsentrasi",
      "Aspek 6 - Spion&Sein","Aspek 7 - Mendahului&Manuver",
      "Aspek 8 - Parkir","Aspek 9 - Rambu&Aturan",
      "Aspek 10 - Sikap&Etika","Total Nilai","Nilai Akhir",
      "Keterangan","Kategori","Kelebihan","Area Perbaikan"]);
  }
  var d = JSON.parse(e.postData.contents);
  sheet.appendRow([
    new Date().toLocaleString("id-ID"),
    d.namaDriver, d.jenisSIM, d.namaPenguji, d.nikPenguji, d.tanggal,
    d.n1, d.n2, d.n3, d.n4, d.n5,
    d.n6, d.n7, d.n8, d.n9, d.n10,
    d.totalNilai, d.nilaiAkhir, d.keterangan, d.kategori,
    d.kelebihan, d.perbaikan
  ]);
  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}
